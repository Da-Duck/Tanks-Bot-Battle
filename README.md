def main():

        sys.argv[1] if len(sys.argv) > 1 else "127.0.0.1",

        if not state:

        enemies = state["enemies"]

        cmd = ""
        
        # –ò–∑–±–µ–≥–∞–µ–º –ø—É–ª–∏

            my_angle = self_tank["angle"]
            
            angle_diff = (enemy_angle - my_angle + 180) % 360 - 180
            
            if angle_diff > 5:
                cmd += "r"
            elif angle_diff < -5:
                cmd += "l"
            else:
                cmd += "s"
        else:
            cmd += "f"
        
        if not cmd:
            cmd = "f"
        
        api.command(cmd)

if __name__ == "__main__":
